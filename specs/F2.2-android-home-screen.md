# F2.2: Android Home Screen - Reminder List

## Objective
Implement the home screen with reminder list, grouped by time.

## Dependencies
- F2.1 (Navigation setup)
- F1.4 (GetRemindersUseCase)

## Requirements
- [ ] ViewModel for home screen with UI state
- [ ] LazyColumn with reminder cards
- [ ] Grouping: Today, Upcoming, Completed
- [ ] Pull-to-refresh (placeholder for sync)
- [ ] Empty state when no reminders
- [ ] FAB to add new reminder

## ViewModel

### presentation/home/HomeViewModel.kt
```kotlin
class HomeViewModel(
    private val getRemindersUseCase: GetRemindersUseCase
) : ViewModel() {

    private val _uiState = MutableStateFlow(HomeUiState())
    val uiState: StateFlow<HomeUiState> = _uiState.asStateFlow()

    init {
        loadReminders()
    }

    private fun loadReminders() {
        viewModelScope.launch {
            getRemindersUseCase(ReminderFilter.All)
                .collect { reminders ->
                    _uiState.update { state ->
                        state.copy(
                            isLoading = false,
                            todayReminders = reminders.filter { it.isToday() },
                            upcomingReminders = reminders.filter { it.isUpcoming() },
                            completedReminders = reminders.filter { it.status == ReminderStatus.COMPLETED }
                        )
                    }
                }
        }
    }

    fun onRefresh() {
        // Placeholder for sync
    }
}

data class HomeUiState(
    val isLoading: Boolean = true,
    val todayReminders: List<Reminder> = emptyList(),
    val upcomingReminders: List<Reminder> = emptyList(),
    val completedReminders: List<Reminder> = emptyList(),
    val error: String? = null
)
```

## Home Screen UI

### presentation/home/HomeScreen.kt
```kotlin
@Composable
fun HomeScreen(
    viewModel: HomeViewModel = koinViewModel(),
    onAddClick: () -> Unit,
    onReminderClick: (String) -> Unit
) {
    val uiState by viewModel.uiState.collectAsState()

    Scaffold(
        floatingActionButton = {
            FloatingActionButton(onClick = onAddClick) {
                Icon(Icons.Default.Add, contentDescription = "Add reminder")
            }
        }
    ) { padding ->
        if (uiState.isLoading) {
            LoadingIndicator()
        } else if (uiState.isEmpty()) {
            EmptyState(
                message = "No reminders yet",
                actionLabel = "Add your first reminder",
                onAction = onAddClick
            )
        } else {
            ReminderList(
                todayReminders = uiState.todayReminders,
                upcomingReminders = uiState.upcomingReminders,
                completedReminders = uiState.completedReminders,
                onReminderClick = onReminderClick,
                modifier = Modifier.padding(padding)
            )
        }
    }
}
```

## Reminder List with Sections

### presentation/home/ReminderList.kt
```kotlin
@Composable
fun ReminderList(
    todayReminders: List<Reminder>,
    upcomingReminders: List<Reminder>,
    completedReminders: List<Reminder>,
    onReminderClick: (String) -> Unit,
    modifier: Modifier = Modifier
) {
    LazyColumn(modifier = modifier) {
        if (todayReminders.isNotEmpty()) {
            item { SectionHeader("Today") }
            items(todayReminders, key = { it.id }) { reminder ->
                ReminderCard(
                    reminder = reminder,
                    onClick = { onReminderClick(reminder.id) }
                )
            }
        }

        if (upcomingReminders.isNotEmpty()) {
            item { SectionHeader("Upcoming") }
            items(upcomingReminders, key = { it.id }) { reminder ->
                ReminderCard(
                    reminder = reminder,
                    onClick = { onReminderClick(reminder.id) }
                )
            }
        }

        if (completedReminders.isNotEmpty()) {
            item { SectionHeader("Completed") }
            items(completedReminders, key = { it.id }) { reminder ->
                ReminderCard(
                    reminder = reminder,
                    onClick = { onReminderClick(reminder.id) }
                )
            }
        }
    }
}
```

## Reminder Card

### presentation/home/ReminderCard.kt
```kotlin
@Composable
fun ReminderCard(
    reminder: Reminder,
    onClick: () -> Unit
) {
    Card(
        modifier = Modifier
            .fillMaxWidth()
            .padding(horizontal = 16.dp, vertical = 4.dp)
            .clickable(onClick = onClick)
    ) {
        Column(modifier = Modifier.padding(16.dp)) {
            Text(
                text = reminder.title,
                style = MaterialTheme.typography.titleMedium
            )
            reminder.notes?.let {
                Text(
                    text = it,
                    style = MaterialTheme.typography.bodyMedium,
                    maxLines = 1,
                    overflow = TextOverflow.Ellipsis
                )
            }
            Text(
                text = reminder.scheduledAt.formatForDisplay(),
                style = MaterialTheme.typography.bodySmall,
                color = MaterialTheme.colorScheme.outline
            )
        }
    }
}
```
